## Role Variables
### defaults/main.yml
<pre><code>
# List of packages
privoxy_packages:
  - privoxy

# Privoxy service name
privoxy_service: privoxy

# privoxy ip
privoxy_ip: "{{ ansible_default_ipv4.address }}"

# privoxy port
privoxy_port: 8118

privoxy_settings:
  "listen-address": "{{ privoxy_ip }}:{{ privoxy_port }}"
</pre></code>
